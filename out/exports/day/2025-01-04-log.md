- **[2025-01-04 18:51] 鄭Force**: import requests
from bs4 import BeautifulSoup
import pandas as pd

# 發送請求獲取網頁內容
url = "https://law.moj.gov.tw/Index.aspx"
response = requests.get(url)
soup = BeautifulSoup(response.content, 'html.parser')

# 爬取熱門法規瀏覽
hot_laws = soup.find('ul', class_='hot-law-list')
hot_laws_data = []
if hot_laws:
    for li in hot_laws.find_all('li'):
        rank = li.find('span', class_='num').text.strip()
        law = li.find('a').text.strip()
        hot_laws_data.append({'排名': rank, '法規名稱': law})

# 爬取最新法規訊息
latest_laws = soup.find('ul', class_='law-list')
latest_laws_data = []
if latest_laws:
    for li in latest_laws.find_all('li'):
        date = li.find('span', class_='date').text.strip()
        category = li.find('span', class_='type').text.strip()
        content = li.find('a').text.strip()
        latest_laws_data.append({'日期': date, '類別': category, '內容': content})

# 創建 DataFrame
df_hot_laws = pd.DataFrame(hot_laws_data)
df_latest_laws = pd.DataFrame(latest_laws_data)

# 顯示結果
print("熱門法規瀏覽：")
print(df_hot_laws)
print("\n最新法規訊息：")
print(df_latest_laws)
